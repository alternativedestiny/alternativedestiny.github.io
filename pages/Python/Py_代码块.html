# 1. 可变进度百分比

import sys

# 输出进度百分比 rate_of_process
# r: 当前步数，p: 总步数，s: 文本
def rop(r, p, s='rate of process'):
    # \r 退格到行首
    str1 = s + ': ' + str(round((r / p) * 100, 2)) + '%'
    sys.stdout.write('\r%s' % str1)
    sys.stdout.flush()
    if r == p:
        sys.stdout.write('\n')

# 4. crc 校验

  1. 代码

    def crc16(x):
    x = x.split()  # 将数据按照空格分开
    a = 0xFFFF  # 初始值
    b = 0xA001  # 校验码
    for byte in x:
        a ^= int(byte, 16)
        for i in range(8):
            last = a % 2
            a >>= 1
            if last == 1:
                a ^= b
    s = hex(a).upper()

    return s[2:6]

if __name__ == "__main__":
    m = input()
    print(crc16(m))

# 7.1. 算法复杂度

算法 最优时间复杂度 平均时间复杂度 最差时间复杂度 最坏空间复杂度
快速排序 O(n log n) O(n log n) O(n²) 不定
冒泡排序 O(n) O(n²) O(n²) O(n)
归并排序 O(n log n) O(n log n) O(n log n) O(n)

# 7.2. 快速排序 (python)

def quick_sort(nums):
    if len(nums) <= 1:
        return nums

    left = []
    right = []
    base = nums.pop()
    for x in nums:
        if x < base:
            left.append(x)
        else:
            right.append(x)

    return quick_sort(left) + [base] + quick_sort(right)

# 7.3. 冒泡排序 (python)

def bubble_sort(nums):
    for j in range(len(nums) - 1):
        for i in range(len(nums) - 1):
            if nums[i] > nums[i+1]:
                nums[i], nums[i+1] = nums[i+1], nums[i]
    return nums
