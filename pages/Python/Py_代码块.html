def quick_sort(nums):
    if len(nums) <= 1:
        return nums

    left = []
    right = []
    base = nums.pop()
    for x in nums:
        if x < base:
            left.append(x)
        else:
            right.append(x)

    return quick_sort(left) + [base] + quick_sort(right)

def bubble_sort(nums):
    for j in range(len(nums) - 1):
        for i in range(len(nums) - 1):
            if nums[i] > nums[i+1]:
                nums[i], nums[i+1] = nums[i+1], nums[i]
    return nums
